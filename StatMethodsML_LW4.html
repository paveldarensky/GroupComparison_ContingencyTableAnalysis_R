library(ggplot2)
library(readr)
library(dplyr)
library(tidyr)  #       

#  
file_path <- "pulse.txt"
pulse_data <- read_delim(file_path, delim="\t", col_names = TRUE)
# delim -    '\t' ()
# col_names = TRUE -   -   ()

#     
# na.omit -    (NA)
data_CB <- na.omit(pulse_data$CB)
data_EB <- na.omit(pulse_data$EB)
data_CA <- na.omit(pulse_data$CA)
data_EA <- na.omit(pulse_data$EA)

#  
cat("  CB:", length(data_CB), "\n")
cat("  EB:", length(data_EB), "\n")
cat("  CA:", length(data_CA), "\n")
cat("  EA:", length(data_EA), "\n")

#     
cat("    CB:", head(data_CB), "\n")
cat("    EB:", head(data_EB), "\n")
cat("    CA:", head(data_CA), "\n")
cat("    EA:", head(data_EA), "\n")

#        
df <- data.frame(
  Pulse = c(data_CB, data_EB, data_CA, data_EA),
  Group = rep(c("CB: ,  ", "EB: ,  ", "CA: ,  ", "EA: ,  "),
              times = c(length(data_CB), length(data_EB), length(data_CA), length(data_EA)))
)

#   
options(repr.plot.width = 12, repr.plot.height = 10)

#   
p1 <- ggplot(df, aes(x = Pulse, fill = Group)) +
  geom_histogram(alpha = 0.6, position = "identity", bins = 15) +
  facet_wrap(~Group) +
  theme_minimal(base_size = 14) +
  ggtitle("  ") +
  theme(legend.position = "none")  #  

print(p1)

# Q-Q plot   
p2 <- ggplot(df, aes(sample = Pulse, color = Group)) +
  stat_qq() +
  stat_qq_line() +
  facet_wrap(~Group) +
  theme_minimal(base_size = 14) +
  ggtitle("  (Q-Q plot)") +
  theme(legend.position = "none")  #  

print(p2)
